88 auto state = _state.load(std::memory_order_acquire);
89 if (state == ResultState::Error) {
90 if (!done) {
91 this->_core.Release()->Done(std::move(_error), [] {

94 _error.~E();
95 } else if (state == ResultState::Exception) {
96 if (!done) {

103
104 void Combine(Result<V, E>&& result) noexcept {
105 auto state = result.State();
106 if (state == ResultState::Value) {
107 return Base::Combine(std::move(result));
108 }
109 auto old_state = _state.load(std::memory_order_acquire);
110 if (old_state == ResultState::Empty &&
111 _state.compare_exchange_strong(old_state, state, std::memory_order_acq_rel, std::memory_order_relaxed)) {
112 if (state == ResultState::Exception) {
113 new (&_exception) std::exception_ptr{std::move(result).Exception()};
114 } else {
115 YACLIB_DEBUG(state != ResultState::Error, "state should be Error, but here it's Empty");
