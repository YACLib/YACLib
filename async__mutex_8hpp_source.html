202 detail::BaseCore* next;
203 if constexpr (Type == UnlockType::Auto) {
204 next = _mutex._waiters;
205 } else {
206 next = _mutex.TryUnlock<FIFO>();
207 if (next == nullptr) {

235 template <bool UnlockFIFO = FIFO>
236 detail::BaseCore* TryUnlock() noexcept {
237 if (_waiters != nullptr) {
238 return _waiters;
239 }
240 const auto old_count_batch_here = std::exchange(_count_batch_here, 0);
241 auto old_state = kLockedNoWaiters;
242 if (_state.load(std::memory_order_relaxed) == old_state &&
243 _state.compare_exchange_strong(old_state, kNotLocked, std::memory_order_release, std::memory_order_relaxed)) {
244 return nullptr;
245 }
246
247 old_state = _state.exchange(kLockedNoWaiters, std::memory_order_acquire);
248 _count_batch_here = old_count_batch_here + 1;
249 if constexpr (UnlockFIFO) {
250 detail::Node* node = reinterpret_cast<detail::BaseCore*>(old_state);
251 detail::Node* prev = nullptr;
252 do {
253 auto* next = node->next;
254 node->next = prev;
255 prev = node;
256 node = next;
257 } while (node != nullptr);
258 return static_cast<detail::BaseCore*>(prev);
259 } else {
260 return reinterpret_cast<detail::BaseCore*>(old_state);
261 }
262 }

265 static constexpr std::uint64_t kLockedNoWaiters = 0;
266 // locked without waiters, not locked, otherwise - head of the awaiters list
