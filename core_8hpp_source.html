44template <typename Ret, typename Arg, typename E, typename Func, CoreType Type, bool kIsAsync, bool kIsCall>
45class Core : public ResultCoreT<Type, Ret, E>, public FuncCore<Func> {
46 using F = FuncCore<Func>;
47 using Storage = typename F::Storage;
48 using Invoke = typename F::Invoke;
49
50 static_assert(!(Type == CoreType::Detach && kIsAsync), "Detach cannot be Async, should be void");
51
52 public:
53 using Base = ResultCoreT<Type, Ret, E>;
54
55 explicit Core(Func&& f) : F{std::forward<Func>(f)} {
56 this->_self = Callback{nullptr, 0};
57 }
58
59 private:
60 void Call() noexcept final {
61 YACLIB_ASSERT(this->_self.unwrapping == 0);
62 if constexpr (Type == CoreType::Run) {
63 YACLIB_ASSERT(this->_self.caller == nullptr);
64 if constexpr (is_invocable_v<Invoke>) {
65 Loop(this, CallImpl<false>(Unit{})); // optimization
66 } else {
67 Loop(this, CallImpl<false>(Result<Arg, E>{Unit{}}));
68 }
69 } else {
70 YACLIB_ASSERT(this->_self.caller != this);
71 auto& core = static_cast<ResultCore<Arg, E>&>(*this->_self.caller);
72 Loop(this, CallImpl<false>(std::move(core.Get())));
73 }
74 }

76 void Drop() noexcept final {
77 Loop(this, CallImpl<false>(Result<Arg, E>{StopTag{}}));

116 template <bool SymmetricTransfer, typename T>
117 [[nodiscard]] YACLIB_INLINE auto CallImpl(T&& r) noexcept try {
118 if constexpr (std::is_same_v<T, Unit> || is_invocable_v<Invoke, Result<Arg, E>>) {
119 return CallResolveAsync<SymmetricTransfer>(std::forward<T>(r));
120 } else {
121 return CallResolveState<SymmetricTransfer>(std::forward<T>(r));
122 }
123 } catch (...) {
124 return Done<SymmetricTransfer>(std::current_exception());
125 }

127 template <bool SymmetricTransfer, bool Async = false, typename T>
128 [[nodiscard]] YACLIB_INLINE auto Done(T&& value) noexcept {
129 // Order defined here is important:
130 // 1. Save caller on stack
131 // 2. Save return value to union where was caller
132 // 3. Destroy and dealloc argument storage
133 // 4. Destroy functor storage
134 // 5. Make current core ready, maybe execute next callback
135 // 3 and 4 can be executed in any order TODO(MBkkt) What order is better here?
136 // Other steps cannot be reordered, examples:
137 // [] (X&& x) -> X&& { touch x, then return it by rvalue reference }
138 // [X x] () -> X&& { touch x, then return it by rvalue reference }
139 auto* caller = this->_self.caller;
140 this->Store(std::forward<T>(value));
141 if constexpr (Type != CoreType::Run || Async) {
142 caller->DecRef();
143 }
144 if constexpr (!Async) {
145 this->_func.storage.~Storage();
146 }
147 return this->template SetResult<SymmetricTransfer>();
148 }

150 template <bool SymmetricTransfer>
151 [[nodiscard]] YACLIB_INLINE auto CallResolveState(Result<Arg, E>&& r) {
152 const auto state = r.State();
153 if constexpr (is_invocable_v<Invoke, Arg> || (std::is_void_v<Arg> && is_invocable_v<Invoke, Unit>)) {
154 if (state == ResultState::Value) {
155 return CallResolveAsync<SymmetricTransfer>(std::move(r).Value());
156 } else if (state == ResultState::Exception) {
157 return Done<SymmetricTransfer>(std::move(r).Exception());
158 } else {
159 YACLIB_ASSERT(state == ResultState::Error);
160 return Done<SymmetricTransfer>(std::move(r).Error());
161 }
162 } else {
163 /**
164 * TLDR: Before and after the "recovery" callback must have the same value type
165 *
166 * Why can't we use the above strategy for other Invoke?
167 * Because user will not have compile error for this case:
168 * MakeFuture()
169 * // Can't call next recovery callback, because our result is Ok, so we will skip next callback
170 * .ThenInline([](E/std::exception_ptr) -> yaclib::Result/Future<double> {
171 * throw std::runtime_error{""};
172 * })
173 * // Previous callback was skipped, so previous Result type is void (received from MakeFuture)
174 * // But here we need Result<double>, so it leeds error
175 * .ThenInline([](yaclib::Result<double>) {
176 * return 1;
177 * });
178 */
179 constexpr bool kIsException = is_invocable_v<Invoke, std::exception_ptr>;
180 constexpr bool kIsError = is_invocable_v<Invoke, E>;
181 static_assert(kIsException ^ kIsError, "Recovery callback should be invokable with std::exception_ptr or E");
182 constexpr auto kState = kIsException ? ResultState::Exception : ResultState::Error;
183 if (state == kState) {
184 using T = std::conditional_t<kIsException, std::exception_ptr, E>;
185 return CallResolveAsync<SymmetricTransfer>(std::get<T>(std::move(r.Internal())));
186 } else {
187 return Done<SymmetricTransfer>(std::move(r));
188 }
189 }
190 }

192 template <bool SymmetricTransfer, typename T>
193 [[nodiscard]] YACLIB_INLINE auto CallResolveAsync(T&& value) {
194 if constexpr (kIsAsync) {
195 auto async = CallResolveVoid(std::forward<T>(value));
196 BaseCore* core = async.GetCore().Release();
197 if constexpr (Type != CoreType::Run) {
198 this->_self.caller->DecRef();
199 this->_self.unwrapping = 1;
200 }
201 this->_self.caller = core;
202 this->_func.storage.~Storage();
203 if constexpr (is_task_v<decltype(async)>) {
204 core->StoreCallback(*this);
205 core = MoveToCaller(core);
206 return Step<SymmetricTransfer>(*this, *core);
207 } else {
208 return core->SetInline<SymmetricTransfer>(*this);
209 }
210 } else {
211 return Done<SymmetricTransfer>(CallResolveVoid(std::forward<T>(value)));
212 }
213 }

215 template <typename T>
216 [[nodiscard]] YACLIB_INLINE auto CallResolveVoid(T&& value) {
217 constexpr bool kArgVoid = is_invocable_v<Invoke>;
218 constexpr bool kRetVoid = std::is_void_v<invoke_t<Invoke, std::conditional_t<kArgVoid, void, T>>>;
219 if constexpr (kRetVoid) {
220 if constexpr (kArgVoid) {
221 std::forward<Invoke>(this->_func.storage)();
222 } else {
223 std::forward<Invoke>(this->_func.storage)(std::forward<T>(value));
224 }
225 return Unit{};
226 } else if constexpr (kArgVoid) {
227 return std::forward<Invoke>(this->_func.storage)();
228 } else {
229 return std::forward<Invoke>(this->_func.storage)(std::forward<T>(value));
230 }
231 }

279template <CoreType CoreT, bool kIsCall, typename Arg, typename E, typename Func>
280auto* MakeCore(Func&& f) {
281 static_assert(CoreT != CoreType::Run || std::is_void_v<Arg>,
282 "It makes no sense to receive some value in first pipeline step");
283 using AsyncRet = result_value_t<typename detail::Return<Arg, E, Func&&>::Type>;
284 static_assert(CoreT != CoreType::Detach || std::is_void_v<AsyncRet>,
285 "It makes no sense to return some value in Detach, since no one will be able to use it");
286 using Ret0 = result_value_t<future_base_value_t<task_value_t<AsyncRet>>>;
287 using Ret = std::conditional_t<std::is_same_v<Ret0, Unit>, void, Ret0>;
288 constexpr bool kIsAsync = is_future_base_v<AsyncRet> || is_task_v<AsyncRet>;
289 // TODO(MBkkt) Think about inline/detach optimization
290 using Core = detail::Core<Ret, Arg, E, Func&&, CoreT, kIsAsync, kIsCall>;
291 return MakeUnique<Core>(std::forward<Func>(f)).Release();
292}

302template <CoreType CoreT, CallbackType CallbackT, typename Arg, typename E, typename Func>
303auto SetCallback(ResultCorePtr<Arg, E>& core, IExecutor* executor, Func&& f) {
304 constexpr bool kIsDetach = CoreT == CoreType::Detach;
305 constexpr bool kIsLazy = CallbackT == CallbackType::LazyInline || CallbackT == CallbackType::LazyOn;
306 constexpr bool kIsCall = CallbackT == CallbackType::On || CallbackT == CallbackType::LazyOn;
307 auto* callback = MakeCore<CoreT, kIsCall, Arg, E>(std::forward<Func>(f));
308 // TODO(MBkkt) callback, executor, caller should be in ctor
309 if constexpr (kIsDetach) {
310 callback->StoreCallback(MakeDrop());
311 }
312 callback->_executor = executor;
313 BaseCore* caller = core.Release();
314 if constexpr (!kIsLazy) {
315 Loop(caller, caller->SetInline<false>(*callback));
316 }
317 using ResultCoreT = typename std::remove_reference_t<decltype(*callback)>::Base;
318 if constexpr (kIsLazy) {
319 callback->next = caller;
320 caller->StoreCallback(*callback);
321 return Task{IntrusivePtr<ResultCoreT>{NoRefTag{}, callback}};
322 } else if constexpr (!kIsDetach) {
323 if constexpr (CallbackT == CallbackType::Inline) {
324 return Future{IntrusivePtr<ResultCoreT>{NoRefTag{}, callback}};
325 } else {
326 return FutureOn{IntrusivePtr<ResultCoreT>{NoRefTag{}, callback}};
327 }
328 }
